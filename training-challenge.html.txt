<script>
const totalWeeks = 6;
const daysPerWeek = 6; // Sunday rest
let progress = 0;

const messages = [
  "Consistency builds strength.",
  "Strong core, strong mind.",
  "Keep going, you're leveling up!",
  "Small steps every day lead to big gains.",
  "Stay consistent, the results will come.",
  "Push yourself, but respect your body."
];

function getRandomMessage() {
  return messages[Math.floor(Math.random() * messages.length)];
}

function createTracker() {
  const container = document.getElementById('weeksContainer');
  for (let w = 1; w <= totalWeeks; w++) {
    const weekDiv = document.createElement('div');
    weekDiv.classList.add('week');
    weekDiv.id = `week${w}`;
    const weekTitle = document.createElement('h2');
    weekTitle.textContent = `Week ${w}`;
    weekDiv.appendChild(weekTitle);

    for (let d = 1; d <= daysPerWeek; d++) {
      const dayDiv = document.createElement('div');
      dayDiv.classList.add('day');
      dayDiv.textContent = d;
      dayDiv.addEventListener('click', () => completeDay(w, d, dayDiv));
      weekDiv.appendChild(dayDiv);
    }

    container.appendChild(weekDiv);
    if (w > 1) weekDiv.style.display = 'none'; // lock future weeks
  }
}

function completeDay(week, day, dayDiv) {
  if (dayDiv.classList.contains('completed')) return;
  dayDiv.classList.add('completed');
  progress++;
  updateProgress();

  const currentWeekDiv = document.getElementById(`week${week}`);
  const completedDays = currentWeekDiv.querySelectorAll('.day.completed').length;
  if (completedDays === daysPerWeek && week < totalWeeks) {
    // unlock next week
    document.getElementById(`week${week + 1}`).style.display = 'block';
    showMessage(`Week ${week} complete! Level Up!`);
  } else {
    showMessage(getRandomMessage());
  }
}

function updateProgress() {
  const totalDays = totalWeeks * daysPerWeek;
  const progressPercent = (progress / totalDays) * 100;
  document.getElementById('progressBar').style.width = progressPercent + '%';
}

function showMessage(msg) {
  const messageDiv = document.getElementById('message');
  messageDiv.textContent = msg;
}

createTracker();
</script>
